    public double findMaxAverage(int[] nums, int k) {
        double sum=0;
        int left=0;
        int right=k-1;
        for(int i=left;i<=right;i++){
            sum+=nums[i];
        }
        double maximum=sum;
        while(right<nums.length-1){
            right++;
            sum+=nums[right];

            sum-=nums[left];

            maximum=Math.max(maximum,sum);
